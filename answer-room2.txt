        function verifyColorOrder() {
            var inputVal = document.getElementById("colorAnswerField").value.trim();
            
            // 💡 【重要】ここに実際の「color-room」の左からの色の正解を入力してください。
            // プレイヤーが打ちそうな表記（ひらがな、カタカナなど）に合わせて変更可能です。
            // 現在は例として「赤青黄」に設定しています。
            const correctAnswer = "赤青黄"; 

            if (inputVal === correctAnswer) {
                alert("……正解だ。封印が解かれた。");
                // 🗺️ 正解したあとの次の移動先
                // 先ほど作った「コード公開ページ（code-release.html）」へ進む場合はここを指定します
                window.location.href = "code-release.html"; 
            } else {
                alert("……記憶が違っているようだ。");
                // 間違えたらゲームオーバー（dark-hazure.htmlなどへ）
                window.location.href = "dark-hazure.html";
            }
        }

        // デベロッパーツール対策
        window.addEventListener('keydown', function (e) {
            if (e.key === 'F12') { e.preventDefault(); return false; }
            if (e.ctrlKey && e.shiftKey && e.key === 'I') { e.preventDefault(); return false; }
            if (e.ctrlKey && e.key === 'u') { e.preventDefault(); return false; }
        });
