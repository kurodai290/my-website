<!-- 💡 bodyの右クリック禁止を、プログラムでON/OFFできるようにIDをつけました -->
<body id="mainBody" oncontextmenu="return isLocked;">

    <!-- 右上の隠しテキストボックス -->
    <div class="hidden-box">
        <input type="text" id="secretInput" placeholder="" onkeydown="checkSecret(event)">
    </div>

    <!-- メイン画面 -->
    <div>
        <div class="title">ふぁんしーなしま</div>
        <p class="warning">※心臓の弱い方はご注意ください</p>
        <a href="room1.html" class="enter-link">はいる</a>
    </div>

    <script>
        // 🔒 初期状態はロックON（true）
        var isLocked = true;

        // 右上の隠しテキストボックスの判定
        function checkSecret(event) {
            if (event.key === 'Enter') {
                var value = document.getElementById("secretInput").value;
                
                // 💡 解除コード「unlockf12」が入力された場合
                if (value === "unlockf12") {
                    isLocked = false; // ロックを解除する
                    alert("デベロッパーツールの制限を解除しました。");
                    document.getElementById("secretInput").value = ""; 
                } 
                // 前回の「おばけ」ルートも残してあります
                else if (value === "おばけ") {
                    alert("秘密の裏口を見つけたね...");
                    window.location.href = "secret-room.html";
                } else {
                    alert("ちがう。");
                    document.getElementById("secretInput").value = "";
                }
            }
        }

        // デベロッパーツール対策のキーボード監視
        window.addEventListener('keydown', function (e) {
            // 🔒 ロックがON（true）の時だけ、F12やショートカットを禁止する
            if (isLocked) {
                if (e.key === 'F12') { e.preventDefault(); return false; }
                if (e.ctrlKey && e.shiftKey && e.key === 'I') { e.preventDefault(); return false; }
                if (e.ctrlKey && e.key === 'u') { e.preventDefault(); return false; }
            }
        });
    </script>
