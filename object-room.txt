    <script>
        function interactDoll(type) {
            const msg = document.getElementById("messageDisplay");
            const flash = document.getElementById("flash-screen");

            if (type === 'blue') {
                // 青の人形：ゲームオーバー処理（お手持ちのdark-hazure.htmlへ）
                flash.style.display = "block";
                setTimeout(() => {
                    alert("青い人形に触れた瞬間、冷たい視線を感じた。");
                    window.location.href = "dark-hazure.html"; 
                }, 100);

            } else if (type === 'yellow') {
                // 黄色の人形：ゲームオーバー処理（お手持ちのdark-hazure.htmlへ）
                flash.style.display = "block";
                setTimeout(() => {
                    alert("黄色い人形が奇妙な声をあげて笑った。");
                    window.location.href = "dark-hazure.html"; 
                }, 100);

            } else if (type === 'correct') {
                // 正解の人形：コード「9130」が出現
                msg.innerText = "人形の背中に剥がれかけのラベルがある……「9130」";
            }
        }

        // デベロッパーツール対策
        window.addEventListener('keydown', function (e) {
            if (e.key === 'F12') { e.preventDefault(); return false; }
            if (e.ctrlKey && e.shiftKey && e.key === 'I') { e.preventDefault(); return false; }
            if (e.ctrlKey && e.key === 'u') { e.preventDefault(); return false; }
        });
