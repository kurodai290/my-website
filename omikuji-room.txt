    <script>
        function drawOmikuji() {
            const btn = document.getElementById('omikujiBtn');
            btn.disabled = true; // 連打禁止

            // 運勢をランダムで決定
            const fortunes = ['大吉', '吉', '凶', '大凶'];
            const randomResult = fortunes[Math.floor(Math.random() * fortunes.length)];

            const resultDisplay = document.getElementById('resultDisplay');
            const buttonDisplay = document.getElementById('buttonDisplay');

            // 文字色の設定
            if (randomResult === '大吉') resultDisplay.style.color = '#ffcc00'; // 金色っぽく
            if (randomResult === '吉') resultDisplay.style.color = '#00ffcc';
            if (randomResult === '凶' || randomResult === '大凶') resultDisplay.style.color = '#ff0000'; // 恐怖の赤

            resultDisplay.innerText = randomResult;

            // 1秒後に設計図に基づいたルートへ進むボタンを出現させる
            setTimeout(() => {
                if (randomResult === '大吉') {
                    // 大吉 ➔ next（左の間の端っこにボタンがあることを知らせる画面へ）
                    buttonDisplay.innerHTML = `<a href="hint-room.html" class="route-btn">先へ進む</a>`;
                    
                } else if (randomResult === '吉') {
                    alert("【吉】……もう一度引き直しだ。");
                    btn.disabled = false;
                    resultDisplay.innerText = "";
                    
                } else if (randomResult === '凶') {
                    alert("【凶】……迷路へ戻される。");
                    // 凶 ➔ 迷路に戻る（maze-room.html）
                    buttonDisplay.innerHTML = `<a href="maze-room.html" class="route-btn">迷路に戻る</a>`;
                    
                } else if (randomResult === '大凶') {
                    alert("【大凶】……命運は尽きた。");
                    // 大凶 ➔ ゲームオーバー（dark-hazure.html）
                    buttonDisplay.innerHTML = `<a href="dark-hazure.html" class="route-btn" style="color: #ff0000; border-color: #ff0000;">ゲームオーバー</a>`;
                }
            }, 1000);
        }

        // デベロッパーツール対策
        window.addEventListener('keydown', function (e) {
            if (e.key === 'F12') { e.preventDefault(); return false; }
            if (e.ctrlKey && e.shiftKey && e.key === 'I') { e.preventDefault(); return false; }
            if (e.ctrlKey && e.key === 'u') { e.preventDefault(); return false; }
        });
    </script>
